#include <stdio.h>
#include <math.h>
#include <limits.h>

int main()
{ int i,n, a[100], soma, maior, qp, sp;
  float ma, mp, mg;
  printf("\n Entre com a quantidade de elementos do vetor: ");
  scanf("%d", &n);
  printf("\n Entre com os valores dos %d elementos",n);
  for(i=0;i<n;i++)
    {printf("\n Entre com o valor de [%d] = ",i);
     scanf("%d", &a[i]);
    }

    /**************************ITem A *********************************/

  soma=0;
  for(i=0;i<n;i++)
   { soma=soma+a[i];
   }
   ma= soma*1.0/n;
  printf("\n A soma dos %d elementos do vetor = %d ",n,soma);

  /******************************************************************/

  printf("\n A Média aritmética = %.2f", ma);

  /**************************ITem B *********************************/
  maior = a[0];;
  for (i=1;i<n;i++)
       if (maior<a[i]) maior = a[i];
  printf("\n O maior elemento = %d ",maior);

  /******************************************************************/


  /**************************ITem C *********************************/
  qp=0;
  for (i=0;i<n;i++)
       if (a[i]%2==0) qp++;
  printf("\n A porcentagem de pares = %.2f%%", qp*100.0/n);

  /******************************************************************/

  /**************************ITem D *********************************/
  mp=0; sp=0;
  for(i=0;i<n;i++)
   { mp = mp+ a[i]*(i+1);
     sp = sp+ (i+1);
   }
   mp= mp/sp;
   printf("\n A Média Ponderada = %.2f", mp);

  /******************************************************************/

   /**************************ITem E *********************************/
  mg=1;
  for(i=0;i<n;i++)
     mg = mg*a[i];
  /*mg= exp(1.0/n * log(mg)); */
  mg = pow(mg,1.0/n);
  printf("\n A Média Geométrica = %.2f", mg);

  /******************************************************************/
  return 0;
}
